local function IsEnemy(p)
    if not p or p == localPlayer then return false end
    if p.Team and localPlayer.Team then return p.Team ~= localPlayer.Team end
    if p.TeamColor and localPlayer.TeamColor then return p.TeamColor ~= localPlayer.TeamColor end
    return true
end

local function RemoveESP(player)
    if ESPStorage[player] then
        for _, obj in pairs(ESPStorage[player]) do
            if typeof(obj) == "Instance" then
                obj:Destroy()
            end
        end
        ESPStorage[player] = nil
    end
    if SkeletonLines[player] then
        for _, line in pairs(SkeletonLines[player]) do
            line:Destroy()
        end
        SkeletonLines[player] = nil
    end
end

local function UpdateSkeleton(player, color, parent)
    if not ESPConfig.Skeleton then
        if SkeletonLines[player] then
            for _, line in pairs(SkeletonLines[player]) do
                line:Destroy()
            end
            SkeletonLines[player] = nil
        end
        return
    end
    
    local char = player.Character
    if not char then return end
    
    if not SkeletonLines[player] then
        SkeletonLines[player] = {}
        for i = 1, #SkeletonParts do
            SkeletonLines[player][i] = CreateSkeletonLine(parent)
        end
    end
    
    for i, connection in ipairs(SkeletonParts) do
        local line = SkeletonLines[player][i]
        local p1 = char:FindFirstChild(connection[1])
        local p2 = char:FindFirstChild(connection[2])
        
        if p1 and p2 then
            local pos1, onScreen1 = camera:WorldToViewportPoint(p1.Position)
            local pos2, onScreen2 = camera:WorldToViewportPoint(p2.Position)
            
            if onScreen1 and onScreen2 then
                local distance = (Vector2.new(pos1.X, pos1.Y) - Vector2.new(pos2.X, pos2.Y)).Magnitude
                local angle = math.atan2(pos2.Y - pos1.Y, pos2.X - pos1.X) * (180 / math.pi)
                
                line.Size = UDim2.new(0, distance, 0, 2)
                line.Position = UDim2.new(0, (pos1.X + pos2.X) / 2, 0, (pos1.Y + pos2.Y) / 2)
                line.Rotation = angle
                line.BackgroundColor3 = color
                line.Visible = ESPConfig.Enabled and ESPConfig.Skeleton
            else
                line.Visible = false
            end
        else
            line.Visible = false
        end
    end
end

local IsShooting = false
local function GetClosest()
    local closest, shortest = nil, math.huge
    local center = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y/2)
    local camPos = camera.CFrame.Position
    
    for _, p in pairs(Players:GetPlayers()) do
        if p == localPlayer then continue end
        if AimbotConfig.TeamCheck and not IsEnemy(p) then continue end
        
        local part = p.Character and p.Character:FindFirstChild(AimbotConfig.AimPart)
        if not part then continue end
        
        local worldDist = (part.Position - camPos).Magnitude
        if worldDist > AimbotConfig.MaxDist then continue end
        
        if FOVConfig.Enabled then
            local screenPos, onScreen = camera:WorldToViewportPoint(part.Position)
            if not onScreen then continue end
            local distFromCenter = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
            if distFromCenter > FOVConfig.Size then continue end
        end
        
        local pos, vis = camera:WorldToViewportPoint(part.Position)
        if not vis then continue end
        
        local screenDist = (Vector2.new(pos.X, pos.Y) - center).Magnitude
        if screenDist < shortest then
            shortest = screenDist
            closest = part
        end
    end
    return closest
end

RunService.RenderStepped:Connect(function()
    -- Ensure UI exists
    EnsureUIExists()
    
    -- Update FOV
    if FOVFrame.Parent then
        FOVFrame.Visible = FOVConfig.Enabled
        FOVFrame.Size = UDim2.new(0, FOVConfig.Size * 2, 0, FOVConfig.Size * 2)
        FOVFrame.Position = UDim2.new(0.5, -FOVConfig.Size, 0.5, -FOVConfig.Size)
    end
    
    -- Update ESP
    if not ESPFolder then return end
    
    for _, p in pairs(Players:GetPlayers()) do
        if p == localPlayer then continue end
        
        if not ESPConfig.Enabled then
            RemoveESP(p)
            continue
        end
        
        local isEnemy = IsEnemy(p)
        if ESPConfig.EnemyOnly and not isEnemy then
            RemoveESP(p)
            continue
        end
        
        local char = p.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if not char or not hum or hum.Health <= 0 or not root then
            RemoveESP(p)
            continue
        end
        
        if not ESPStorage[p] then CreateESP(p, ESPFolder) end
        local esp = ESPStorage[p]
        
        local myRoot = localPlayer.Character and localPlayer.Character:FindFirstChild("HumanoidRootPart")
        local dist = myRoot and (root.Position - myRoot.Position).Magnitude or 0
        if dist > ESPConfig.MaxDist then
            for _, obj in pairs(esp) do
                if typeof(obj) == "Instance" then obj.Visible = false end
            end
            UpdateSkeleton(p, isEnemy and Color3.fromRGB(255, 60, 60) or Color3.fromRGB(60, 255, 100), ESPFolder)
            continue
        end
        
        -- Get character bounding box
        local cf, size = char:GetBoundingBox()
        local topY = (cf * CFrame.new(0, size.Y/2, 0)).Position
        local bottomY = (cf * CFrame.new(0, -size.Y/2, 0)).Position
        
        local topScreen, topVis = camera:WorldToViewportPoint(topY)
        local bottomScreen, bottomVis = camera:WorldToViewportPoint(bottomY)
        
        if not topVis and not bottomVis then
            for _, obj in pairs(esp) do
                if typeof(obj) == "Instance" then obj.Visible = false end
            end
            UpdateSkeleton(p, isEnemy and Color3.fromRGB(255, 60, 60) or Color3.fromRGB(60, 255, 100), ESPFolder)
            continue
        end
        
        local boxHeight = math.abs(topScreen.Y - bottomScreen.Y)
        local boxWidth = boxHeight * 0.6
        local centerX = (topScreen.X + bottomScreen.X) / 2
        local centerY = (topScreen.Y + bottomScreen.Y) / 2
        
        local color = isEnemy and Color3.fromRGB(255, 60, 60) or Color3.fromRGB(60, 255, 100)
        
        -- Box
        local boxVisible = ESPConfig.Boxes
        esp.BoxTop.Visible = boxVisible
        esp.BoxBottom.Visible = boxVisible
        esp.BoxLeft.Visible = boxVisible
        esp.BoxRight.Visible = boxVisible
        
        if boxVisible then
            local thickness = 2
            esp.BoxTop.Size = UDim2.new(0, boxWidth, 0, thickness)
            esp.BoxTop.Position = UDim2.new(0, centerX - boxWidth/2, 0, centerY - boxHeight/2)
            esp.BoxTop.BackgroundColor3 = color
            
            esp.BoxBottom.Size = UDim2.new(0, boxWidth, 0, thickness)
            esp.BoxBottom.Position = UDim2.new(0, centerX - boxWidth/2, 0, centerY + boxHeight/2 - thickness)
            esp.BoxBottom.BackgroundColor3 = color
            
            esp.BoxLeft.Size = UDim2.new(0, thickness, 0, boxHeight)
            esp.BoxLeft.Position = UDim2.new(0, centerX - boxWidth/2, 0, centerY - boxHeight/2)
            esp.BoxLeft.BackgroundColor3 = color
            
            esp.BoxRight.Size = UDim2.new(0, thickness, 0, boxHeight)
            esp.BoxRight.Position = UDim2.new(0, centerX + boxWidth/2 - thickness, 0, centerY - boxHeight/2)
            esp.BoxRight.BackgroundColor3 = color
        end
        
        -- Name
        esp.Name.Visible = ESPConfig.Names
        esp.Name.Text = p.DisplayName
        esp.Name.TextColor3 = color
        esp.Name.Size = UDim2.new(0, 100, 0, 15)
        esp.Name.Position = UDim2.new(0, centerX - 50, 0, centerY - boxHeight/2 - 20)
        
        -- Distance
        esp.Dist.Visible = ESPConfig.Distance
        esp.Dist.Text = string.format("%.0fm", dist)
        esp.Dist.TextColor3 = Color3.fromRGB(200, 200, 200)
        esp.Dist.Size = UDim2.new(0, 100, 0, 12)
        esp.Dist.Position = UDim2.new(0, centerX - 50, 0, centerY + boxHeight/2 + 2)
        
        -- Health
        esp.HPBG.Visible = ESPConfig.Health
        esp.HP.Visible = ESPConfig.Health
        if ESPConfig.Health then
            local hpPct = hum.Health / hum.MaxHealth
            local barHeight = boxHeight - 4
            esp.HPBG.Size = UDim2.new(0, 4, 0, barHeight)
            esp.HPBG.Position = UDim2.new(0, centerX - boxWidth/2 - 8, 0, centerY - barHeight/2)
            
            esp.HP.Size = UDim2.new(1, 0, hpPct, 0)
            esp.HP.Position = UDim2.new(0, 0, 1 - hpPct, 0)
            esp.HP.BackgroundColor3 = Color3.fromRGB(255 * (1-hpPct), 255 * hpPct, 60)
        end
        
        UpdateSkeleton(p, color, ESPFolder)
    end
end)

-- ============================================
RunService.RenderStepped:Connect(function()
    if AimbotConfig.Enabled and IsShooting then
        local target = GetClosest()
        if target then
            local pos = camera:WorldToViewportPoint(target.Position)
            local dx = (pos.X - camera.ViewportSize.X/2) / AimbotConfig.Smoothness
            local dy = (pos.Y - camera.ViewportSize.Y/2) / AimbotConfig.Smoothness
            pcall(function() if mousemoverel then mousemoverel(dx, dy) end end)
        end
    end
    
    if SilentAim.Enabled and IsShooting then
        local closest, shortest = nil, math.huge
        local center = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y/2)
        
        for _, p in pairs(Players:GetPlayers()) do
            if p == localPlayer then continue end
            local head = p.Character and p.Character:FindFirstChild("Head")
            if not head then continue end
            local pos, vis = camera:WorldToViewportPoint(head.Position)
            if not vis then continue end
            
            if FOVConfig.Enabled then
                local distFromCenter = (Vector2.new(pos.X, pos.Y) - center).Magnitude
                if distFromCenter > FOVConfig.Size then continue end
            end
            
            local dist = (Vector2.new(pos.X, pos.Y) - center).Magnitude
            if dist < shortest and dist < 250 then
                shortest = dist
                closest = head
            end
        end
        
        if closest then
            pcall(function() camera.CFrame = CFrame.new(camera.CFrame.Position, closest.Position) end)
        end
    end
end)

RunService.Heartbeat:Connect(function()
    if Speed.Enabled and localPlayer.Character then
        local hum = localPlayer.Character:FindFirstChildOfClass("Humanoid")
        local root = localPlayer.Character:FindFirstChild("HumanoidRootPart")
        if hum and root and hum.MoveDirection.Magnitude > 0 then
            root.Velocity = hum.MoveDirection * Speed.Value + Vector3.new(0, root.Velocity.Y, 0)
        end
    end
    
    if BHOP.Enabled and localPlayer.Character then
        local hum = localPlayer.Character:FindFirstChildOfClass("Humanoid")
        if hum and hum:GetState() == Enum.HumanoidStateType.Landed then
            hum:ChangeState(Enum.HumanoidStateType.Jumping)
        end
    end
    
    if Fly.Enabled and Fly.BodyV and Fly.BodyG then
        local vel = Vector3.new(0, 0, 0)
        local cf = camera.CFrame
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then vel = vel + cf.LookVector * Fly.Speed end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then vel = vel - cf.LookVector * Fly.Speed end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then vel = vel - cf.RightVector * Fly.Speed end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then vel = vel + cf.RightVector * Fly.Speed end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then vel = vel + Vector3.new(0, Fly.Speed, 0) end
        Fly.BodyV.Velocity = vel
        Fly.BodyG.CFrame = cf
    end
    
    if Noclip.Enabled and localPlayer.Character then
        for _, p in pairs(localPlayer.Character:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide = false end
        end
    end
end)

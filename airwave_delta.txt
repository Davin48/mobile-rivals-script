local function IsEnemy(player)
    if not player or player == localPlayer then return false end
    if player.Team and localPlayer.Team then
        return player.Team ~= localPlayer.Team
    end
    if player.TeamColor and localPlayer.TeamColor then
        return player.TeamColor ~= localPlayer.TeamColor
    end
    return true
end

local function RemoveESP(player)
    if ESPStorage[player] then
        if ESPStorage[player].billboard then
            ESPStorage[player].billboard:Destroy()
        end
        ESPStorage[player] = nil
    end
end

-- ESP Update Loop
RunService.RenderStepped:Connect(function()
    for _, player in pairs(Players:GetPlayers()) do
        if player == localPlayer then continue end
        
        if not ESPConfig.Enabled then
            RemoveESP(player)
            continue
        end
        
        local isEnemy = IsEnemy(player)
        
        -- Enemy Only Check
        if ESPConfig.EnemyOnly and not isEnemy then
            RemoveESP(player)
            continue
        end
        
        local char = player.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        local head = char and char:FindFirstChild("Head")
        local humanoid = char and char:FindFirstChildOfClass("Humanoid")
        
        if not char or not root or not humanoid then
            RemoveESP(player)
            continue
        end
        
        if humanoid.Health <= 0 then
            RemoveESP(player)
            continue
        end
        
        if not ESPStorage[player] then
            CreateESP(player)
        end
        
        local esp = ESPStorage[player]
        esp.billboard.Adornee = root
        
        -- Color
        local enemyColor = Color3.fromRGB(255, 60, 60)
        local teamColor = Color3.fromRGB(60, 255, 100)
        local color = isEnemy and enemyColor or teamColor
        
        -- Distance
        local myRoot = localPlayer.Character and localPlayer.Character:FindFirstChild("HumanoidRootPart")
        local dist = myRoot and (root.Position - myRoot.Position).Magnitude or 0
        
        if dist > ESPConfig.MaxDistance then
            esp.billboard.Enabled = false
            continue
        end
        
        esp.billboard.Enabled = true
        
        -- Update Box
        esp.box.Visible = ESPConfig.Boxes
        esp.box.BackgroundColor3 = color
        esp.box.BorderColor3 = color
        
        -- Update Name
        esp.name.Visible = ESPConfig.Names
        esp.name.TextColor3 = color
        
        -- Update Distance
        esp.distance.Visible = ESPConfig.Distance
        esp.distance.Text = string.format("%.0fm", dist)
        
        -- Update Health
        esp.healthBG.Visible = ESPConfig.Health
        esp.healthBar.Visible = ESPConfig.Health
        local hpPercent = humanoid.Health / humanoid.MaxHealth
        esp.healthBar.Size = UDim2.new(hpPercent, 0, 1, 0)
        esp.healthBar.BackgroundColor3 = Color3.fromRGB(
            255 * (1 - hpPercent),
            255 * hpPercent,
            60
        )
    end
end)

local AimbotConfig = {
    Enabled = false,
    TeamCheck = true,
    AimPart = "Head",
    MaxDistance = 200,
    Smoothness = 2,
    AutoFire = false,
}

local function GetClosestPlayer()
    local closest = nil
    local shortestDist = math.huge
    local center = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y / 2)
    local camPos = camera.CFrame.Position
    
    for _, player in pairs(Players:GetPlayers()) do
        if player == localPlayer then continue end
        
        if AimbotConfig.TeamCheck then
            local isSameTeam = false
            if player.Team and localPlayer.Team then
                isSameTeam = player.Team == localPlayer.Team
            elseif player.TeamColor and localPlayer.TeamColor then
                isSameTeam = player.TeamColor == localPlayer.TeamColor
            end
            if isSameTeam then continue end
        end
        
        local char = player.Character
        local part = char and char:FindFirstChild(AimbotConfig.AimPart)
        if not part then continue end
        
        local worldDist = (part.Position - camPos).Magnitude
        if worldDist > AimbotConfig.MaxDistance then continue end
        
        local screenPos, visible = camera:WorldToViewportPoint(part.Position)
        if not visible then continue end
        
        local screenDist = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
        if screenDist < shortestDist then
            shortestDist = screenDist
            closest = part
        end
    end
    
    return closest
end

-- Aimbot Loop
RunService.RenderStepped:Connect(function()
    if not AimbotConfig.Enabled then return end
    
    local target = GetClosestPlayer()
    if target then
        local pos = camera:WorldToViewportPoint(target.Position)
        local centerX = camera.ViewportSize.X / 2
        local centerY = camera.ViewportSize.Y / 2
        
        local deltaX = (pos.X - centerX) / AimbotConfig.Smoothness
        local deltaY = (pos.Y - centerY) / AimbotConfig.Smoothness
        
        -- Delta compatible mouse movement
        pcall(function()
            if mousemoverel then
                mousemoverel(deltaX, deltaY)
            end
        end)
        
        if AimbotConfig.AutoFire then
            pcall(mouse1click)
        end
    end
end)

-- SILENT AIM (Auto-lock nearest)
-- ============================================
local SilentAimEnabled = false
local SilentAimTarget = nil

local function GetSilentAimTarget()
    local closest = nil
    local shortestDist = math.huge
    local center = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y / 2)
    
    for _, player in pairs(Players:GetPlayers()) do
        if player == localPlayer then continue end
        local char = player.Character
        local head = char and char:FindFirstChild("Head")
        if not head then continue end
        
        local screenPos, visible = camera:WorldToViewportPoint(head.Position)
        if not visible then continue end
        
        local dist = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
        if dist < shortestDist and dist < 250 then
            shortestDist = dist
            closest = head
        end
    end
    return closest
end

RunService.Heartbeat:Connect(function()
    if SilentAimEnabled then
        SilentAimTarget = GetSilentAimTarget()
        if SilentAimTarget then
            pcall(function()
                camera.CFrame = CFrame.new(camera.CFrame.Position, SilentAimTarget.Position)
            end)
        end
    end
end)

local Fly = {Enabled = false, Speed = 50, BodyG = nil, BodyV = nil}

local function StartFly()
    if not localPlayer.Character then return end
    local root = localPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    
    if Fly.BodyG then Fly.BodyG:Destroy() end
    if Fly.BodyV then Fly.BodyV:Destroy() end
    
    Fly.BodyG = Instance.new("BodyGyro")
    Fly.BodyG.P = 9e4
    Fly.BodyG.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    Fly.BodyG.CFrame = root.CFrame
    Fly.BodyG.Parent = root
    
    Fly.BodyV = Instance.new("BodyVelocity")
    Fly.BodyV.Velocity = Vector3.new(0, 0, 0)
    Fly.BodyV.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    Fly.BodyV.Parent = root
end

local function StopFly()
    if Fly.BodyG then Fly.BodyG:Destroy() end
    if Fly.BodyV then Fly.BodyV:Destroy() end
    Fly.BodyG = nil
    Fly.BodyV = nil
end

RunService.Heartbeat:Connect(function()
    if Fly.Enabled and Fly.BodyV and Fly.BodyG then
        local camCF = camera.CFrame
        local vel = Vector3.new(0, 0, 0)
        
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then
            vel = vel + camCF.LookVector * Fly.Speed
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then
            vel = vel - camCF.LookVector * Fly.Speed
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then
            vel = vel - camCF.RightVector * Fly.Speed
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then
            vel = vel + camCF.RightVector * Fly.Speed
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
            vel = vel + Vector3.new(0, Fly.Speed, 0)
        end
        
        Fly.BodyV.Velocity = vel
        Fly.BodyG.CFrame = camCF
    end
end)

    AddSlider("Max Distance", 50, 200, 1000, function(v) AimbotConfig.MaxDistance = v end)
    AddSlider("Smoothness", 1, 2, 10, function(v) AimbotConfig.Smoothness = v end)
    
    -- SILENT AIM
    AddSection("SILENT AIM")
    AddToggle("Enable Silent Aim", false, function(v) SilentAimEnabled = v end)
    
    -- ESP SECTION
    AddSection("ESP")
    AddToggle("Enable ESP", false, function(v) ESPConfig.Enabled = v end)
    AddToggle("Boxes", true, function(v) ESPConfig.Boxes = v end)
    AddToggle("Names", true, function(v) ESPConfig.Names = v end)
    AddToggle("Distance", true, function(v) ESPConfig.Distance = v end)
    AddToggle("Health Bar", true, function(v) ESPConfig.Health = v end)
    AddToggle("Enemy Only", false, function(v) ESPConfig.EnemyOnly = v end)
    AddSlider("Max Distance", 100, 1000, 5000, function(v) ESPConfig.MaxDistance = v end)
    
    -- MISC SECTION
    AddSection("MISC")
    AddToggle("Speedhack", false, function(v) Speedhack.Enabled = v end)
    AddSlider("Speed Value", 16, 32, 300, function(v) Speedhack.Speed = v end)
    AddToggle("Bunny Hop", false, function(v) BHOP.Enabled = v end)
    AddToggle("Fly", false, function(v)
        Fly.Enabled = v
        if v then StartFly() else StopFly() end
    end)
    AddSlider("Fly Speed", 10, 50, 200, function(v) Fly.Speed = v end)
    AddToggle("NoClip", false, function(v)
        Noclip.Enabled = v
        if v then
            Noclip.Connection = RunService.Stepped:Connect(function()
                for _, part in pairs(localPlayer.Character:GetDescendants()) do
                    if part:IsA("BasePart") then part.CanCollide = false end
                end
            end)
        else
            if Noclip.Connection then Noclip.Connection:Disconnect() end
            for _, part in pairs(localPlayer.Character:GetDescendants()) do
                if part:IsA("BasePart") then part.CanCollide = true end
            end
        end
    end)

        TeleportService:Teleport(17625359962, localPlayer)
    end)

    -- Hide/Show
    local hidden = false
    local openBtn = nil
    
    hideBtn.MouseButton1Click:Connect(function()
        hidden = not hidden
        if hidden then
            mainFrame.Size = UDim2.new(0, 300, 0, 40)
            content.Visible = false
            hideBtn.Text = "+"
        else
            mainFrame.Size = UDim2.new(0, 300, 0, 450)
            content.Visible = true
            hideBtn.Text = "-"
        end
    end)
    
    closeBtn.MouseButton1Click:Connect(function()
        screenGui.Enabled = false
        
        -- Create open button
        if not openBtn then
            openBtn = Instance.new("TextButton")
            openBtn.Size = UDim2.new(0, 60, 0, 60)
            openBtn.Position = UDim2.new(0, 10, 0.5, -30)
            openBtn.BackgroundColor3 = Color3.fromRGB(60, 120, 180)
            openBtn.Text = "☰"
            openBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
            openBtn.TextSize = 24
            openBtn.Font = Enum.Font.GothamBold
            openBtn.Parent = screenGui
            Instance.new("UICorner", openBtn).CornerRadius = UDim.new(1, 0)
            openBtn.Draggable = true
            
            openBtn.MouseButton1Click:Connect(function()
                screenGui.Enabled = true
                openBtn.Visible = false
            end)
        else
            openBtn.Visible = true
        end
    end)
end
